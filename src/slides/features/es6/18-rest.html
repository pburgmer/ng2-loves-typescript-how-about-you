Rest Parameters
  • Einsatzgebiet: Variable Parameterlisten
    • Statt arguments Variable
  • Argumente als echtes Array entgegennehmen
    • Impliziter Type any[]
    • Explizit angeben z.B. number[]
function fn(a, b, ...derRest) {
  console.log(derRest);
}

fn(1, 2); // []
fn(1, 2, 3); // [3]
fn(1, 2, 3, 4); // [3, 4]
