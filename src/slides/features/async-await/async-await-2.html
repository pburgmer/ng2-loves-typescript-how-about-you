async/await 
function check() {
  return new Promise<boolean>(resolve => resolve(false));
}

function doNotGiveIn() {
  return new Promise<void>(resolve => { console.log("Do it now!"); resolve(); });
}

async function askForCleanup() {
  console.log("Please clean up your room");
  while(await !check()) { await doNotGiveIn(); }
  console.log("Thanks!");
}

askForCleanup();
