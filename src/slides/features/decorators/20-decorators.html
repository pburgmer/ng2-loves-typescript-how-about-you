Decorators
  • Können verwendet werden an
    • Klassen (Konstruktor)
    • Properties & Methoden
    • Parametern
  • Müssen immer (auch ohne Parameter) aufrufende Klammern haben ()
@Freezed()
class Person {

  constructor(@NotUndefined() private name:string) {}

  @LogCall("warn")
  greet() { console.log(`Hello, I'm ${this.name}`); }
}
